<script>
(function () {
  const grid = document.getElementById('articlesGrid');
  if (!grid) return;

  const BATCH = 6; 
  let shown = 0;
  let currentFilter = 'all';

  const catMap = {
    'stocks': '\u0627\u0644\u0623\u0633\u0647\u0645',
    'realestate': '\u0627\u0644\u0639\u0642\u0627\u0631\u0627\u062a',
    'gold': '\u0627\u0644\u0630\u0647\u0628 \u0648\u0627\u0644\u0645\u0639\u0627\u062f\u0646',
    'crypto': '\u0627\u0644\u0639\u0645\u0644\u0627\u062a \u0627\u0644\u0631\u0642\u0645\u064a\u0647',
    'startup': '\u0631\u064a\u0627\u062f\u0629 \u0627\u0644\u0623\u0639\u0645\u0627\u0644',
    'passive': '\u0627\u0644\u062f\u062e\u0644 \u0627\u0644\u0633\u0644\u0628\u064a',
    'personal': '\u0627\u0644\u0645\u0627\u0644\u064a\u0647 \u0627\u0644\u0634\u062e\u0635\u064a\u0647',
    'tech': '\u062a\u0642\u0646\u064a\u0647 \u0648\u062a\u0632\u0648\u062f'
  };

  function getFilteredArticles() {
    if (typeof ARTICLES === 'undefined') return [];
    if (currentFilter === 'all') return ARTICLES;
    const targetCat = catMap[currentFilter];
    return ARTICLES.filter(a => a.category === targetCat);
  }

  function renderArticles(append = false) {
    if (!append) { grid.innerHTML = ''; shown = 0; }
    const filtered = getFilteredArticles();
    const slice = filtered.slice(shown, shown + BATCH);
    
    slice.forEach(art => {
      const card = document.createElement('article');
      card.className = 'article-card';
      card.innerHTML = `
        <img src="${art.image}" alt="${art.title}">
        <div class="article-content">
          <div class="article-meta">
            <span class="article-category">${art.category}</span>
            <span>${art.date}</span>
          </div>
          <h3>${art.title}</h3>
          <p>${art.excerpt}</p>
          <a href="article.html?id=${art.id}" class="read-more">
            \u0627\u0642\u0631\u0623 \u0627\u0644\u0645\u0632\u064a\u062f
          </a>
        </div>
      `;
      grid.appendChild(card);
    });
    
    shown += slice.length;
    const loadMoreBtn = document.getElementById('loadMoreBtn');
    if (loadMoreBtn) {
      loadMoreBtn.style.display = (shown >= filtered.length) ? 'none' : 'block';
    }
  }

  // Event listeners
  document.addEventListener('click', (e) => {
    if (e.target.classList.contains('filter-btn')) {
      const filterBar = document.getElementById('categoryFilter');
      filterBar.querySelector('.active').classList.remove('active');
      e.target.classList.add('active');
      currentFilter = e.target.dataset.category;
      renderArticles();
    }
    if (e.id === 'loadMoreBtn') {
       renderArticles(true);
    }
  });

  const loadMoreBtn = document.getElementById('loadMoreBtn');
  if (loadMoreBtn) {
    loadMoreBtn.onclick = () => renderArticles(true);
  }

  // Initial load - wait for everything
  window.addEventListener('load', () => {
     renderArticles();
  });
})();
</script>
</body>
</html>
